Exercice : Programme en C

#include stdio.h
int factorielle(int n) {
return (n == 0 || n == 1) ? 1 : n * factorielle(n - 1);
}
int main() {
int nombre;
printf("Entrez un nombre : ");
scanf("%d", &nombre);
printf("La factorielle de %d est : %d\n", nombre, factorielle(nombre));
return 0;
}
